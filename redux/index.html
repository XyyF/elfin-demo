Redux

独立的数据管理框架，不仅仅是在React中。极简的大小，仅有2KB。

数据流程必定是：某个Actions → 通过Dispatch广播 → 在Reducer中返回新的State

reducer

如果在reducer中直接修改state，是无法监听到state变化的

/**
 * state 数据，因为第一次初始化没有action注意初始化
 * action dispatch广播的行为
 */
function reducer(state = 0, action) {};
  
var store = Redux.createStore(reducer);

store.subscribe

通过store进行订阅注册，当store变化时，触发回调

store.subscribe(function() {});

Redux.bindActionCreators

将Action函数和store.dispatch进行绑定后返回新的广播函数，好处是广播时无需关注store在哪儿 

function incrementAction() {
    return { type: 'INCREMENT' };
};
const increment = Redux.bindActionCreators(incrementAction, store.dispatch);

Redux.combineReducers

组件多个reducer为一个store

function todoer(state = []) { return state };
  
// { counter: 0, todoer: [] }
Redux.createStore(
  Redux.combineReducers(counter, todoer)
);
